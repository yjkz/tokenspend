- [ ] **Step 3: Create LanguageContext with Provider and hook**

```typescript
// src/i18n/index.ts
import { createContext, useContext, useState, useCallback, type ReactNode } from 'react'
import zh from './locales/zh'
import en from './locales/en'

type Lang = 'zh' | 'en'
type Translations = typeof zh

const translations: Record<Lang, Translations> = { zh, en }

interface LanguageContextValue {
  lang: Lang
  setLang: (lang: Lang) => void
  t: (key: keyof Translations, params?: Record<string, string | number>) => string
}

const LanguageContext = createContext<LanguageContextValue | null>(null)

export function LanguageProvider({ children }: { children: ReactNode }) {
  const [lang, setLangState] = useState<Lang>(() => {
    const saved = localStorage.getItem('lang')
    return saved === 'en' || saved === 'zh' ? saved : 'zh'
  })

  const setLang = useCallback((newLang: Lang) => {
    setLangState(newLang)
    localStorage.setItem('lang', newLang)
  }, [])

  const t = useCallback((key: keyof Translations, params?: Record<string, string | number>) => {
    let text = translations[lang][key] || key
    if (params) {
      for (const [k, v] of Object.entries(params)) {
        text = text.replace(new RegExp(`\\{${k}\\}`, 'g'), String(v))
      }
    }
    return text
  }, [lang])

  return (
    <LanguageContext.Provider value={{ lang, setLang, t }}>
      {children}
    </LanguageContext.Provider>
  )
}

export function useLanguage() {
  const ctx = useContext(LanguageContext)
  if (!ctx) throw new Error('useLanguage must be used within LanguageProvider')
  return ctx
}
```

### Task 3: Add Language Switcher to Sidebar

**Files:**
- Modify: `src/components/layout/AppLayout.tsx`

- [ ] **Step 1: Read current AppLayout.tsx**

- [ ] **Step 2: Replace with i18n version**

```tsx
import { Outlet, NavLink } from 'react-router-dom'
import { useLanguage } from '../../i18n'

export default function AppLayout() {
  const { lang, setLang, t } = useLanguage()

  const links = [
    { to: '/', label: t('dashboard') },
    { to: '/sessions', label: t('sessions') },
  ]

  return (
    <div className="flex h-screen">
      <aside className="w-56 bg-gray-900 border-r border-gray-800 flex flex-col">
        <div className="px-5 py-4 border-b border-gray-800">
          <h1 className="text-lg font-bold text-white">{t('appTitle')}</h1>
        </div>
        <nav className="flex-1 p-3 space-y-1">
          {links.map(l => (
            <NavLink
              key={l.to}
              to={l.to}
              end={l.to === '/'}
              className={({ isActive }) =>
                `block px-3 py-2 rounded text-sm transition-colors ${
                  isActive
                    ? 'bg-gray-800 text-white'
                    : 'text-gray-400 hover:bg-gray-800/50 hover:text-gray-200'
                }`
              }
            >
              {l.label}
            </NavLink>
          ))}
        </nav>
        <div className="px-4 py-3 border-t border-gray-800 flex items-center gap-2">
          <span className="text-gray-500 text-xs">🌐</span>
          <button
            onClick={() => setLang('zh')}
            className={`text-xs px-2 py-1 rounded transition-colors ${
              lang === 'zh' ? 'bg-gray-700 text-white' : 'text-gray-500 hover:text-gray-300'
            }`}
          >
            中文
          </button>
          <button
            onClick={() => setLang('en')}
            className={`text-xs px-2 py-1 rounded transition-colors ${
              lang === 'en' ? 'bg-gray-700 text-white' : 'text-gray-500 hover:text-gray-300'
            }`}
          >
            English
          </button>
        </div>
      </aside>
      <main className="flex-1 overflow-auto p-6">
        <Outlet />
      </main>
    </div>
  )
}
```
